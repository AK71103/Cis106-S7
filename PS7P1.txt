def IntCalc():
    while True:
        try:
            PrnAmnt = float(input("Enter Principle Amount: ")) #enter Principle Amount and Interest Rate
            if PrnAmnt<=0:
                print("Principal must be greater than zero.")
                continue

            IntRate = float(input("Enter Interest Rate : "))
            if IntRate<0:
                print("Interest rate can't be negative.")
                continue
        except ValueError:
            print("Invalid input. Please enter numeric values.")
            continue

        #Processing
        YrCnt= range(1,6) #1st five years #$
        IntDec= (IntRate/100) #convert rate to Decimal #$
        print("\nYears|Begining Balance|Ending Balance|Total Interest") #$
        print("_ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _ _") #$
        
        BgnBal = (PrnAmnt) #The begining balance is the Principle #$
        Annint= ((PrnAmnt) * (IntDec)) #The Annual Interest is the Principle times Interest Rate. #$
        TotInt= 0.0

        for x in YrCnt:
            
            AnnInt = ((BgnBal) * (IntDec)) #The annual interest is the begining balance times the interest rate. #$
            EndBal = ((BgnBal) + (AnnInt)) #The ending balance is the begining balance plus the annual interest. #$
            TotInt += AnnInt #Total Interest is the added total of 5 years of annual interest. #$
           
            print(f"{x:<4} | ${BgnBal:>16,.2f} | ${AnnInt:>9,.2f} | ${EndBal:>14,.2f}") #Displays Year number,Begining Balance, Ending Balance,Total Interest #$
            
            BgnBal = EndBal # Update Begining Balance for next Year #$
            
            print(f"Total accumulated interest over 5 years: ${TotInt:,.2f}")
             
             # Ask if user wants to run again
            choice = input("Do you want to calculate again? (y/n): ")
            if choice != 'y':
                print("Program ended.")
                break
 
   # Run the program
if __name__ == "__main__":
    IntCalc()
